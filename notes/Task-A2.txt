--  	1- agents:

		CREATE TABLE agents (
			acode          SERIAL PRIMARY KEY,
			fname          VARCHAR(255),
			lname          VARCHAR(255),
			dob            DATE,
			gender         VARCHAR(10),
			nationality    VARCHAR(80),
			security_level INT,
			agent_type     VARCHAR(255),
			agent_status   VARCHAR(50)
		);


--		2- skills:

		CREATE TABLE skills (
			skill_code   SERIAL PRIMARY KEY,
			skill_type   VARCHAR(255),
			skill_level  INT
		);


--		3- targets:
		
		CREATE TABLE targets (
			tid           SERIAL PRIMARY KEY,
			ttype         VARCHAR(60),
			tdescription  TEXT
		);


--		4- assets:

		CREATE TABLE assets (
			asid     SERIAL PRIMARY KEY,
			astype   VARCHAR(100),
			asmake   VARCHAR(100),
			asmodel  VARCHAR(255),
			astag  VARCHAR(255),
			asstatus VARCHAR(80)
		);


--		5- contacts:

		CREATE TABLE contacts (
			acode          INT,
			phone          VARCHAR(25),
			email          VARCHAR(255),
			address_line1  VARCHAR(255),
			city           VARCHAR(255),
			country        VARCHAR(255),
			from_date      DATE,
			to_date        DATE,
			FOREIGN KEY (acode) REFERENCES agents(acode)
		);


--		6- secret_identities:

		CREATE TABLE secret_identities (
			sid                SERIAL PRIMARY KEY,
			sid_type           VARCHAR(255),
			sid_nationality    VARCHAR(255),
			sid_other_details  VARCHAR(255),
			sid_status         VARCHAR(100),
			sid_issue_date     DATE,
			sid_exp_date       DATE,
			acode INT,
			FOREIGN KEY (acode) REFERENCES agents(acode)
		);


--		7- missions:

		CREATE TABLE missions (
			mid                 SERIAL PRIMARY KEY,
			general_description TEXT,
			city                VARCHAR(255),
			country             VARCHAR(255),
			start_date          DATE,
			end_date            DATE,
			mstatus             VARCHAR(100),
			tid                 INT,
			FOREIGN KEY (tid) REFERENCES targets(tid)
		);


--		8- has: 

		CREATE TABLE has (
			acode      INT,
			skill_code INT,
			FOREIGN KEY (acode) REFERENCES agents(acode),
			FOREIGN KEY (skill_code) REFERENCES skills(skill_code)
		);


--		9- participate_in:

		CREATE TABLE participate_in (
			acode INT,
			mid   INT,
			FOREIGN KEY (acode) REFERENCES agents(acode),
			FOREIGN KEY (mid) REFERENCES missions(mid)
		);


--		10- used_in:

		CREATE TABLE used_in (
			asid INT,
			mid  INT,
			FOREIGN KEY (asid) REFERENCES assets(asid),
			FOREIGN KEY (mid) REFERENCES missions(mid)
		);
